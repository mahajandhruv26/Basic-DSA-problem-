#include <iostream>
#include <vector>

int main() {
    std::vector<int> numbers = {1, 2, 3, 4, 5};

    // Modify each element using reference
    for (int& num : numbers) {
        num *= 2; // doubles the value in-place
    }

    // Print updated vector
    for (const int& num : numbers) {      output:
        std::cout << num << " ";          //2 4 6 8 10
    }

    return 0;
}
