#include<iostream>
using namespace std;
int arr(int ar[],int n){
    for(int i=0;i<n;i++){
        cin>>ar[i];
    }
    return 0;
}
void print(int ar[],int n){
    cout<<"{";
    for(int i=0;i<n;i++){
        cout<<ar[i];
        if(i!=n-1){
            cout<<",";
        }
    }
    cout<<"}";
}
int unique(int arr[],int n,int uniq[]){
    int k=0;
    for(int i=0;i<n;i++){
        bool m=1;
        for(int j=0;j<n;j++){
            if(i==j){
                continue;
            }
            if(arr[i]==arr[j]){
                m=0;
                break;
            }
        }
        if(m){
            uniq[k]=arr[i];
            k++;}
    }return k;
}
int main(){
    int aary[100],n,un[100];
    cout<<"enter size of first aaray:";
    cin>>n;
    arr(aary,n);
    cout<<"array:";
    print(aary,n);
    int k=unique(aary,n,un);
    cout<<endl<<"the unique array:";
    print(un,k);
}
