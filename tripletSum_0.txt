       for(int i=0;i<n;i++){

}
void triplet(int arr[],int n){

	int c[3];
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			for(int k=j+1;k<n;k++){
				if(arr[i]+arr[j]+arr[k]==0){
					cout<<arr[i]<<" ";
					cout<<arr[j]<<" ";
					cout<<arr[k];

				}
			}
		}
	}

}
