#include <bits/stdc++.h> 
vector<int> findArraySum(vector<int>&a, int n, vector<int>&b, int m) {
	// Write your code here.
 int c=0;//sum =s and carry=c
vector<int> res;
n--;
m--;
 while(m>=0 && n>=0){
     int s=a[n]+b[m]+c;
     c=s/10;
     s=s%10;
     res.push_back(s);
     m--;
     n--;
 }
 while(m>=0){
     int s=b[m]+c;
     c=s/10;
     s=s%10;
     res.push_back(s);
     m--;
 }
 while(n>=0){
     int s=a[n]+c;
     c=s/10;
     s=s%10;
     res.push_back(s);
     n--;
 }
 if(c){
     res.push_back(c);
 }
 reverse(res.begin(),res.end());
return res;
}
