#include<iostream>
using namespace std;
int arr(int ar[],int n){
    for(int i=0;i<n;i++){                       
        cin>>ar[i];
    }
    return 0;
}
void print(int ar[],int n){
    for(int i=0;i<n;i++){
        cout<<ar[i]<<" ";
    }
    cout<<endl;
}
int swp(int arr[],int n){
    int a=0,b=1;
    while(b<n){                                      
        swap(arr[a],arr[b]);
        a+=2;
        b+=2;
    } 
    return 0;
}
int main(){
    int aary[100],n;
    cin>>n;
    arr(aary,n);
    print(aary,n);
    swp(aary,n);
    cout<<endl<<"the swap array is:-"<<endl;
    print(aary,n);
}
