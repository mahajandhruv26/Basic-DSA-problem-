#include<iostream>
#include<math.h>
using namespace std;
bool prime(int n){
    if(n==1 || n==0){
        return false;
    }
    if (n==2){
        return true;
    }
    for(int i=2;i*i<=n;i++){ //for not using i<sqrt(n)
        if(n%i==0){
            return false;
        }
        
    }
    return true;
}
int sum(int n){
    int res=0;
    while(n){
        res+=n%10;
        n/=10;
    }
    return res;
}
int digi(int n){
    int c=0;
    int a=1;
    while(c<n){
        a++;
        if(prime(a)){
            c++;
        }
        //return a; it will give the mth prime no
    }
    int add=0;
    add=sum(a);
    while(add>=10){
        add=sum(add);
        }
    return add*a;
}
    
int main(){
    int n,m;
    cin>>m>>n;
    cout<<digi(m);
